\textbf{R}
\begin{lstlisting}[style=rstyle]
library(readxl); library(ggfortify)
re <- read_excel("real_estate.xlsx")
num_cols <- re |>
  select(price, area, bedrooms,
    bathrooms, year_built)
pca <- prcomp(num_cols, scale. = TRUE)
autoplot(pca, data = re,
  colour = "type",
  loadings = TRUE,
  loadings.label = TRUE) +
  theme_minimal()
\end{lstlisting}
\end{column}
\begin{column}{0.48\textwidth}
\textbf{Python}
\begin{lstlisting}[style=pythonstyle]
from sklearn.decomposition import PCA
from sklearn.preprocessing import (
    StandardScaler)
re = pd.read_excel("real_estate.xlsx")
X = StandardScaler().fit_transform(
    re[["price","area","bedrooms",
        "bathrooms","year_built"]])
pca = PCA(n_components=2)
scores = pca.fit_transform(X)
plt.scatter(scores[:,0], scores[:,1],
    c=re["type"].map(
        {"house":"#1565C0",
         "apartment":"#E53935"}),
    s=10, alpha=0.5)
\end{lstlisting}
\end{column}
\end{columns}

library(pheatmap)
corr <- cor(num_cols, use = "complete")
pheatmap(corr,
  color = colorRampPalette(

  )(100),
  display_numbers = TRUE,

# Parallel coordinates
library(GGally)
ggparcoord(re,
  columns = c("price","area",
    "bedrooms","bathrooms"),
  groupColumn = "type",
  scale = "uniminmax",
  alphaLines = 0.1) +
  scale_color_manual(values =
    c(house="#1565C0",
      apartment="#E53935"))

\begin{column}{0.48\textwidth}
\textbf{Python}
\begin{lstlisting}[style=pythonstyle]
# Clustered heatmap
sns.clustermap(
    re[num_cols].corr(),
    annot=True, fmt=".2f",
    cmap="RdBu_r", vmin=-1, vmax=1,
    figsize=(7, 6))
# Parallel coordinates
from pandas.plotting import (
    parallel_coordinates)
cols = ["price","area","bedrooms",
        "bathrooms","type"]
df_norm = re[cols].copy()
for c in cols[:-1]:
    df_norm[c] = (df_norm[c] -
        df_norm[c].min()) / (
        df_norm[c].max() -
        df_norm[c].min())
parallel_coordinates(df_norm, "type",
    colormap="coolwarm", alpha=0.1)
\end{lstlisting}

\textbf{R (leaflet)}
\begin{lstlisting}[style=rstyle]
library(leaflet)
pal <- colorNumeric("YlOrRd",
  domain = re$price)
leaflet(re) |>
  addProviderTiles(
    "CartoDB.Positron") |>
  addCircleMarkers(
    lng = ~lon, lat = ~lat,
    radius = 3,
    color = ~pal(price),
    fillOpacity = 0.7,
    popup = ~paste0(
      "Price: $", price, "<br>",
      "Area: ", area, " sqm")) |>
  addLegend(pal = pal,
    values = ~price,
    title = "Price ($)")
\end{lstlisting}
\end{column}
\begin{column}{0.48\textwidth}
\textbf{Python (folium)}
\begin{lstlisting}[style=pythonstyle]
import folium
m = folium.Map(
    location=[re.lat.mean(),
              re.lon.mean()],
    zoom_start=12,
    tiles="CartoDB positron")
for _, r in re.iterrows():
    folium.CircleMarker(
        [r["lat"], r["lon"]],
        radius=3,
        color="red",
        fill=True,
        fill_opacity=0.7,
        popup=(f"${r['price']:,.0f}"
               f"<br>{r['area']} sqm")
    ).add_to(m)
m.save("real_estate_map.html")
\end{lstlisting}
