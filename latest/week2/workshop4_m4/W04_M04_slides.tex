\section{Why Missing Data Matters}

\begin{frame}{The Silent Problem}
  Most visualization tools \textbf{silently drop} missing rows:
  \begin{itemize}
    \item \texttt{ggplot2}: ``Removed 847 rows containing missing values''
    \item \texttt{matplotlib}: NaN values produce gaps or are skipped
    \item \texttt{seaborn}: drops NA before computing statistics
  \end{itemize}
  \vspace{6pt}
  If 15\% of your data is missing and it is \textbf{not random}, every
  chart you produce is biased --- but you will never know unless you
  explicitly visualize the missingness itself.
  \begin{alertblock}{Pro-Tip}
    Rule: \textbf{before any EDA plot, first plot the missingness}.
    This is Step 0 of any analysis. If you skip it, you are making
    invisible assumptions about which data you are not seeing.
  \end{alertblock}
\end{frame}

\section{Three Mechanisms}

\begin{frame}{MCAR vs MAR vs MNAR (Rubin, 1976)}
  \begin{center}
    \includegraphics[width=0.95\textwidth]{figures_w04m04/missing_mechanisms}
  \end{center}
\end{frame}

\begin{frame}{How to Diagnose the Mechanism}
  \centering
  \begin{tabular}{lp{5.5cm}l}
    \toprule
    \textbf{Mechanism} & \textbf{Diagnostic} & \textbf{Tool} \\
    \midrule
    MCAR & Compare distributions of observed data between
           ``has NA'' vs ``no NA'' groups. No difference $\Rightarrow$ MCAR.
         & Little's test, t-tests \\
    MAR  & Missingness in column A correlates with values in
           column B. A depends on observed B.
         & Missingness dendrogram, logistic regression \\
    MNAR & Can't fully test --- but if paid apps hide ratings
           more than free apps, and you can't predict it from
           other columns, suspect MNAR.
         & Domain knowledge, sensitivity analysis \\
    \bottomrule
  \end{tabular}
  \vspace{6pt}
  \begin{alertblock}{Pro-Tip}
    MNAR is the most dangerous: the missing values are systematically
    different from observed values. No imputation method fully fixes
    MNAR bias --- you must acknowledge the limitation.
  \end{alertblock}
\end{frame}

# Missingness matrix

msno.bar(apps, figsize=(8,4))

(apps.isna().mean()
    .sort_values(ascending=False)

\section{Shadow-Augmented Plots}

\begin{frame}{Making the Invisible Visible}
  \begin{center}
    \includegraphics[width=0.95\textwidth]{figures_w04m04/shadow}
  \end{center}
\end{frame}

\begin{frame}[fragile]{The Shadow Matrix (naniar)}
  \begin{columns}[T]
    \begin{column}{0.48\textwidth}
      \textbf{R (naniar)}
\begin{lstlisting}[style=rstyle]
# geom_miss_point: shows NA rows
# at the margin of a scatter plot
ggplot(apps, aes(x = Size,
                 y = Rating)) +
  geom_miss_point(alpha = 0.3) +
  theme_minimal() +
  labs(title = "Shadow Scatter")

# bind_shadow: add _NA columns
apps_shadow <- apps |>
  bind_shadow()  # adds Rating_NA etc.

# Then facet by missingness
ggplot(apps_shadow, aes(
    x = Reviews, fill = Rating_NA)) +
  geom_histogram(bins = 30,
    position = "identity",
    alpha = 0.5) +
  scale_fill_manual(values =
    c("!NA" = "#1565C0",
      "NA"  = "#E53935"))
\end{lstlisting}
    \end{column}
    \begin{column}{0.48\textwidth}
      \textbf{Concept}
      \vspace{4pt}

      The \textbf{shadow matrix} creates a parallel Boolean DataFrame
      where each cell records ``is this value missing?''

      \vspace{4pt}
      \texttt{bind\_shadow()} (R) or
      \texttt{df.isna().add\_suffix("\_NA")} (Python) appends
      these columns, enabling:

      \vspace{4pt}
      \begin{itemize}
        \item Colour by missingness in scatter plots
        \item Facet histograms by ``missing vs present''
        \item Test whether the distribution of $X$ differs
              between ``$Y$ missing'' vs ``$Y$ present'' groups
      \end{itemize}

      \vspace{4pt}
      If distributions differ $\Rightarrow$ \textbf{MAR} or \textbf{MNAR}.
    \end{column}
  \end{columns}
\end{frame}

\section{Co-occurrence Patterns}

\begin{frame}{UpSet Plot: Which Columns Go Missing Together?}
  \begin{center}
    \includegraphics[width=0.82\textwidth]{figures_w04m04/upset}
  \end{center}
  \begin{exampleblock}{Data Insight}
    The UpSet pattern reveals \textbf{cascading missingness}: when Size
    is missing, Reviews is often missing too (18 co-occurrences). This
    suggests a shared data-entry failure, not independent missingness.
    In R: \texttt{gg\_miss\_upset(apps)}. In Python: \texttt{missingno}
    or \texttt{upsetplot} package.
  \end{exampleblock}
\end{frame}

\begin{frame}{Missingness Dendrogram}
  \begin{center}
    \includegraphics[width=0.62\textwidth]{figures_w04m04/dendrogram}
  \end{center}
  \begin{alertblock}{Pro-Tip}
    The dendrogram clusters columns by the \textbf{correlation of their
    missingness indicators}. Columns that cluster together tend to go
    missing in the same rows. In R: \texttt{naniar} or manual
    \texttt{hclust()}. In Python: \texttt{msno.dendrogram()}.
  \end{alertblock}
\end{frame}

  \end{center}

\begin{frame}{Imputation Strategies at a Glance}
  \centering
  \begin{tabular}{lp{4.5cm}l}
    \toprule
    \textbf{Method} & \textbf{Effect on Distribution} & \textbf{When OK} \\
    \midrule
    Drop rows & Reduces $n$; biased if not MCAR & MCAR, few NAs \\
    Mean/median & Spike at centre; shrinks variance & Never ideal \\
    Mode (categorical) & Inflates modal category & Rare categories \\
    Regression & Preserves trend; underestimates variance & MAR \\
    Multiple (mice/MICE) & Multiple plausible datasets; correct CI & MAR, gold standard \\
    \bottomrule
  \end{tabular}
  \vspace{6pt}
  \begin{alertblock}{Pro-Tip}
    Always \textbf{plot the distribution before and after imputation}
    (overlay histograms, Q-Q plots). If the imputed distribution looks
    different from the observed, document the discrepancy.
    In R: \texttt{mice} package. In Python: \texttt{sklearn.impute}
    or \texttt{miceforest}.
  \end{alertblock}
\end{frame}

\section{Synthesis}

\begin{frame}{Key Takeaways}
  \begin{enumerate}
    \item \textbf{Plot the missingness before any analysis}. It is Step 0.
          Standard charts silently drop NA rows, hiding potential bias.
    \item \textbf{Three mechanisms} (Rubin, 1976): MCAR (safe to drop),
          MAR (impute from observed), MNAR (bias guaranteed).
    \item The \textbf{missingness matrix} reveals vertical stripes (one
          bad column), horizontal bands (bad rows), and block patterns
          (structured missingness).
    \item \textbf{Shadow-augmented plots} colour or facet by missingness,
          revealing whether missing rows differ from observed rows.
    \item \textbf{UpSet plots} and \textbf{dendrograms} identify
          co-occurring missingness (cascading data-entry failures).
    \item \textbf{Always visualize imputation effects}: overlay
          before/after distributions to detect artificial spikes.
  \end{enumerate}
\end{frame}
