  \begin{alertblock}{Pro-Tip}
    Netflix has two unique EDA challenges: (1) multi-valued columns
    require \texttt{separate\_rows()} / \texttt{.explode()} before
    counting; (2) ``duration'' means minutes for movies but seasons
    for TV shows --- never analyse them together without splitting.
  \end{alertblock}

\begin{lstlisting}[style=pythonstyle]
nf = pd.read_csv("netflix.csv")
nf["date_added"] = pd.to_datetime(
    nf["date_added"].str.strip(),
    errors="coerce")
nf["year_added"] = (
    nf["date_added"].dt.year)

# Split duration by type
nf["dur_min"] = (nf
    .loc[nf["type"]=="Movie",
         "duration"]
    .str.extract(r"(\d+)")
    .astype(float))
nf["seasons"] = (nf
    .loc[nf["type"]=="TV Show",
         "duration"]
    .str.extract(r"(\d+)")
    .astype(float))

nf["primary_genre"] = (nf["listed_in"]
    .str.split(",").str[0].str.strip())

# Genre explosion
nf_genres = (nf.assign(
    genre=nf["listed_in"]
    .str.split(", ")).explode("genre"))
\end{lstlisting}

\section{Key Findings}

\begin{frame}{Four Key Findings}
  \begin{center}
    \includegraphics[width=0.95\textwidth]{figures_w04m10/findings}
  \end{center}
\end{frame}

\begin{frame}{Finding 1: The TV Pivot}
  \begin{itemize}
    \item Movie additions peaked around 2017, then declined
    \item TV Show additions grew steadily through 2020--2021
    \item This mirrors Netflix's public strategy shift toward original
          series (``bingeable'' content)
    \item \textbf{Implication}: any time-series analysis should account
          for this structural break. Pre-2016 and post-2016 are
          different populations.
  \end{itemize}
\end{frame}

\begin{frame}{Finding 2: Director Missingness is Structural (MAR)}
  \begin{itemize}
    \item 30\% of titles have no director listed
    \item TV Shows are far more likely to have missing directors than
          Movies (TV often has multiple directors per episode)
    \item This is \textbf{MAR}: missingness depends on the observed
          variable ``type''
    \item \textbf{Implication}: analyse director only for Movies;
          for TV Shows, encode \texttt{has\_director} as a binary
          feature or use the show's creator instead
  \end{itemize}
\end{frame}

\begin{frame}{Finding 3: Duration Is Near-Normal for Movies}
  \begin{itemize}
    \item Mean $\approx$ 100 minutes, SD $\approx$ 28 minutes
    \item Histogram is approximately symmetric, slight right tail
    \item Q-Q plot follows the diagonal reasonably well
    \item \textbf{Implication}: parametric methods (t-tests, ANOVA) are
          appropriate for movie duration comparisons. No log transform
          needed (unlike Reviews in Google Play).
  \end{itemize}
\end{frame}

\begin{frame}{Finding 4: Extreme Geographic Concentration}
  \begin{itemize}
    \item US produces 37\% of Netflix content; India 14\%
    \item Top 2 countries = 51\% of all titles
    \item Long tail: 100+ countries each produce $<$1\%
    \item \textbf{Implication}: group rare countries into ``Other'' for
          any categorical analysis. Use \texttt{fct\_lump\_n()} (R) or
          \texttt{.where(isin(top\_n))} (Python).
  \end{itemize}
\end{frame}
